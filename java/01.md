サンプル　じゃんけん

```java
import java.util.Random;  //util=utility　汎用的に使われるクラス
import java.util.Scanner;  //Scanner:キーボード入力を読み込む、Random:乱数を作る

public class Janken {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in); //キーボード入力を読む準備
        Random random = new Random(); //Random型(クラス名)の変数random(変数)を作る
                                    　//new Random():Randomクラスのインスタンス(実体)を作る

        while (true) {
            System.out.println("じゃんけんをしてください");
            System.out.println("0: グー  1: チョキ  2: パー");

            int player = scanner.nextInt(); //入力された数字を読む
            int computer = random.nextInt(3);

            System.out.println("あなた: " + hand(player));
            System.out.println("相手: " + hand(computer));

            if (player == computer) {
                System.out.println("あいこ");
            } else if (
                (player == 0 && computer == 1) ||
                (player == 1 && computer == 2) ||
                (player == 2 && computer == 0)
            ) {
                System.out.println("勝ち");
                break;
            } else {
                System.out.println("負け");
                break;
            }
        }

        scanner.close(); //Scannerの使用終了
    }

    static String hand(int num) {
        switch (num) {
            case 0:
                return "グー";
            case 1:
                return "チョキ";
            case 2:
                return "パー";
            default:
                return "不正入力";
        }
    }
}

```
